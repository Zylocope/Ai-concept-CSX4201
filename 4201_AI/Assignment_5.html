    kmeans = KMeans(n_clusters=i, init='k-means++', max_iter=300, n_init=10,

Step 5: Interpret Results

K-mean clustering aim data into k groups. Each data point is assigned to the cluster with the nearest centroid (cluster center).The elbow method is often used to determine the optimal number of clusters by identifying the "elbow point" in the WCSS plot.In our data we use cluster 5. DBSCAN is a density-based clustering algorithm. It groups together data points that are closely packed together (points with many nearby neighbors), marking as outliers points that lie alone in low-density regions (whose nearest neighbors are too far away).In our data we use radius 0.19 and minimum sample 9. Pro of k-means is simple and efficient for large datasets. But it's cons is sensitive to the initial placement of centroids. It require specifying the number of clusters before implementing. DBSCAN's pros are robust to outliers, and does not require specifying the numbers of clusters. But it is sensitive to the choice of parameters.It also has the con of performance degrade with high-dismensional data.

import heapq

class Node:
    def __init__(self, state, parent=None, g=0, h=0):
        self.state = state
        self.parent = parent
        self.g = g
        self.h = h
        self.f = g + h

    def __lt__(self, other):
        return self.f < other.f


def manhattan_distance(start_state, goal_state):
  """Calculates Manhattan distance heuristic."""
  distance = 0
  for num in range(1, 9):
    for i in range(3):
      for j in range(3):
        if start_state[i][j] == num:
          x1, y1 = i, j
        if goal_state[i][j] == num:
            x2, y2 = i, j
    distance += abs(x1 - x2) + abs(y1 - y2)
  return distance

def get_neighbors(state):
    """Finds possible neighbor states by swapping the 0 tile."""
    neighbors = []
    zero_row, zero_col = next((r, c) for r, row in enumerate(state)
                             for c, val in enumerate(row) if val == 0)

    for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
        new_row, new_col = zero_row + dr, zero_col + dc
        if 0 <= new_row < 3 and 0 <= new_col < 3:
            new_state = [list(row) for row in state]
            new_state[zero_row][zero_col], new_state[new_row][new_col] = new_state[new_row][new_col], new_state[zero_row][zero_col]
            neighbors.append(tuple(tuple(row) for row in new_state))
    return neighbors

def print_board(state):
  for row in state:
      print(" ".join(map(str, [x if x != 0 else " " for x in row ]))) # Replace 0 with blank space.
  print()

def astar(start, goal):
    """Performs A* search to find the shortest path."""
    open_set = []
    closed_set = set()

    start_node = Node(start, g=0, h=manhattan_distance(start, goal))
    heapq.heappush(open_set, start_node)

    while open_set:
        current_node = heapq.heappop(open_set)

        if current_node.state == goal:
            path = []
            cost = current_node.g
            while current_node:
              path.append(current_node.state)
              current_node = current_node.parent
            return path[::-1],cost # Return reversed path

        closed_set.add(current_node.state)

        for neighbor_state in get_neighbors(current_node.state):
            if neighbor_state in closed_set:
                continue

            neighbor_node = Node(neighbor_state, current_node, current_node.g + 1, manhattan_distance(neighbor_state, goal))

            if neighbor_node not in open_set :
                heapq.heappush(open_set, neighbor_node)

    return None, None  # No path found

def main():
    """Main function to demonstrate the A* algorithm."""

    start = ((1, 2, 3), (4, 0, 6), (7, 5, 8))
    end = ((1, 2, 3), (4, 5, 6), (7, 8, 0))

    path, total_cost = astar(start, end)

    if path:
        print(f"Solution found in {len(path) - 1} moves with total cost {total_cost}:\n")
        for i, board in enumerate(path):
            print(f"Move {i}:")
            print_board(board)
    else:
        print("No solution found.")

if __name__ == "__main__":
    main()
